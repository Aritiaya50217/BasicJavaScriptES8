คำสั่ง While

เป็นคำสั่งที่ทำซ้ำไปเรื่อยๆ จนกว่าเงื่อนไขจะเป็นเท็จ ในคำสั่งเราจะต้องกำหนดเงื่อนไข โดยจะออกจากคำสั่ง While ทันทีเมื่อเงื่อนไขเป็นเท็จ

var i = 1 ;
while (i <= 10) {
   console.log(i++)
}

  • เหมือนคำสั่ง while แต่จะทำคำสั่งใน {...} ก่อนหนึ่งรอบจึงค่อยตรวจสอบเงื่อนไข

    • i = 1 ;

       console.log(i);
       i++ ;

    Infinite Loop

    ถ้าเงื่อนไขไม่เป็นเท็จ จะเกิดการทำซ้ำไม่รู้จบ

    break And continue

  • break ถ้าเจอคำสั่งนี้หยุดทำซ้ำทันที เช่น
          for (i = 1; i <= 10; i++){
             if (i ==3) break ;
             console.log(i);
          }
          ผลลัพธ์
          1
          2
  • continue ถ้าเจอคำสั่งนี้จะข้ามไปทำรอบใหม่ทันที เช่น
          for (i = 1; i <= 10; i++){
             if (i ==3) continue ;
             console.log(i);
          }
          ผลลัพธ์
          1
          2
          4
          5
          6
          7
          8
